    /**
 * annotation-layer.js — Main orchestrator.
 * Updated for pdf.js v4.1.392
 *
 * Lifecycle is entirely event-bus driven:
 *   - pagerendered   → attach overlays, render annotations
 *   - scalechanging  → re-check overlays
 *   - resize         → re-check overlays
 *   - updateviewarea → track current page (for future use)
 *   - sidebarviewchanged → sync our pane state with native sidebar
 *   - cursortoolchanged  → sync if user changes tool via pdfjs UI
 *
 * The layer never programmatically clicks pdfjs buttons.
 * Instead it dispatches events through the eventBus.
 */

import * as T from './templates.js';
import { ToolManager, defaultTools } from './tools.js';
import { buildStorage } from './storage.js';

// ─── Helpers ──────────────────────────────────────────────────

function clamp(v, lo, hi) { return Math.max(lo, Math.min(hi, v)); }
function pixelToNorm(pageDiv, px, py) {
  const r = pageDiv.getBoundingClientRect();
  return { x: clamp((px - r.left) / r.width, 0, 1), y: clamp((py - r.top) / r.height, 0, 1) };
}
function genId() {
  return 'ann_' + Date.now().toString(36) + '_' + Math.random().toString(36).slice(2, 8);
}

// ─── Constants ────────────────────────────────────────────────
// pdf.js 4.1.392 sidebar view enum (matches SidebarView in pdf_sidebar.js)
const SIDEBAR_VIEW = { NONE: 0, THUMBS: 1, OUTLINE: 2, ATTACHMENTS: 3, LAYERS: 4 };
const CURSOR_TOOL = { SELECT: 0, HAND: 1 };

// ─── AnnotationLayer ──────────────────────────────────────────

export default class AnnotationLayer {
  constructor(config) {
    this.config = {
      apiBaseUrl: '/api/annotations',
      documentId: null,
      currentUser: { name: 'Reviewer', initials: 'R' },
      tools: null, // pass custom tools array to extend/replace
      ...config,
    };

    this.annotations = new Map();
    this.overlays = new Map();       // pageNumber → overlay div
    this.elements = new Map();       // annotation id → DOM element on page
    this.cardElements = new Map();   // annotation id → card in the pane

    this.paneOpen = false;
    this.textFilter = '';
    this.categoryFilter = '';
    this.selectedId = null;
    this.pendingForm = null;         // { el, opts } while a new-comment form is open
    this.currentPage = 1;
    this._openingSidebar = false;    // guard: suppress sidebarviewchanged during our toggle clicks

    this.store = buildStorage(this.config);

    // Tool system
    const tools = this.config.tools || defaultTools();
    this.toolMgr = new ToolManager(tools, this);
  }

  // ── Public init ─────────────────────────────────────────────

  async init() {
    this.app = await this._waitForViewer();
    this.eventBus = this.app.eventBus;

    await this._waitForFirstPage();

    this._injectCSS();
    this._disableFeatures();
    this._buildFloatingToolbar();
    this._buildSidebarPane();
    this._bindEventBus();
    this._bindKeyboard();

    // Initial overlay pass
    this._ensureOverlays();

    // Load saved annotations
    try {
      const saved = await this.store.load();
      if (Array.isArray(saved)) {
        saved.forEach(a => { a._persisted = true; this.annotations.set(a.id, a); });
        this._renderAll();
        this._rebuildList();
      }
    } catch (e) {
      console.warn('[AnnotationLayer] Load failed:', e);
    }

    // Set initial tool
    this.toolMgr.setActive('pointer');
    this._syncToolbarUI();
  }

  // ── Event Bus Bindings (the core of the architecture) ───────

  _bindEventBus() {
    const eb = this.eventBus;

    // Page rendered → attach overlay, render annotations for that page
    eb.on('pagerendered', ({ pageNumber }) => {
      requestAnimationFrame(() => {
        this._ensureOverlays();
        this._renderAnnotationsForPage(pageNumber);
      });
    });

    // All pages loaded → full overlay pass
    eb.on('pagesloaded', () => {
      requestAnimationFrame(() => this._ensureOverlays());
    });

    // New document loaded → clear previous annotations and load fresh
    eb.on('documentloaded', () => {
      this._clearAll();
      this.store.load().then(saved => {
        if (Array.isArray(saved)) {
          saved.forEach(a => { a._persisted = true; this.annotations.set(a.id, a); });
          this._renderAll();
          this._rebuildList();
        }
      }).catch(e => console.warn('[AnnotationLayer] Reload failed:', e));
    });

    // Pages destroyed (e.g. switching documents) → clean up overlays
    eb.on('pagesdestroy', () => {
      this.overlays.forEach(o => o.remove());
      this.overlays.clear();
      this.elements.clear();
    });

    // Scale / resize → ensure overlays exist (debounced)
    let scaleTimer = null;
    const onScaleOrResize = () => {
      if (scaleTimer) cancelAnimationFrame(scaleTimer);
      scaleTimer = requestAnimationFrame(() => { scaleTimer = null; this._ensureOverlays(); });
    };
    eb.on('scalechanging', onScaleOrResize);
    eb.on('resize', onScaleOrResize);

    // Track current page
    eb.on('updateviewarea', ({ location }) => {
      if (location && location.pageNumber) {
        this.currentPage = location.pageNumber;
      }
    });

    // Sidebar view changed — sync our pane visibility.
    // In 4.1.392 the sidebar fires sidebarviewchanged with { view } where
    // view is one of the SIDEBAR_VIEW constants.
    eb.on('sidebarviewchanged', ({ view }) => {
      if (this._openingSidebar) return; // we caused this, ignore it

      if (view === SIDEBAR_VIEW.NONE) {
        // Sidebar was closed entirely
        this._onPaneClosed();
      } else if (this.paneOpen) {
        // A native view was selected while our pane was showing — hide ours
        this._onPaneClosed();
      }
    });

    // Cursor tool changed by pdfjs (e.g. user picks hand tool from secondary toolbar)
    eb.on('cursortoolchanged', ({ tool }) => {
      if (tool === CURSOR_TOOL.HAND && this.toolMgr.activeTool?.id !== 'hand') {
        this._cancelForm();
        this.toolMgr.activeTool = this.toolMgr.get('hand');
        this._syncToolbarUI();
      } else if (tool === CURSOR_TOOL.SELECT && this.toolMgr.activeTool?.id === 'hand') {
        this.toolMgr.activeTool = this.toolMgr.get('pointer');
        this._syncToolbarUI();
      }
    });

    // Clean up stale draw state on any global mouseup
    document.addEventListener('mouseup', () => {
      const hl = this.toolMgr.get('highlight');
      if (hl && hl._drawing) {
        if (hl._drawing.el) hl._drawing.el.remove();
        hl._drawing = null;
      }
    });
  }

  // ── Dispatch events into pdfjs (instead of clicking buttons) ─

  dispatchCursorTool(toolId) {
    this.eventBus.dispatch('switchcursortool', { source: this, tool: toolId });
  }

  // ── CSS injection ───────────────────────────────────────────

  _injectCSS() {
    if (document.getElementById('ann-layer-css')) return;
    const link = document.createElement('link');
    link.id = 'ann-layer-css';
    link.rel = 'stylesheet';
    const cssPath = this.config.cssPath || 'src/pdfjs-annotations.css';
    link.href = cssPath;
    document.head.appendChild(link);
  }

  // ── Disable unwanted pdfjs features ─────────────────────────

  _disableFeatures() {
    // Hide annotation editor UI — ids that exist in 4.1.392
    const hide = [
      'editorFreeText', 'editorInk', 'editorStamp', 'editorHighlight',
      'editorModeButtons', 'editorModeSeparator',
      'pageRotateCw', 'pageRotateCcw',
      'secondaryOpenFile', 'presentationMode',
      'viewBookmark', 'viewBookmarkSeparator',
    ];
    hide.forEach(id => {
      const el = document.getElementById(id);
      if (el) { el.style.display = 'none'; el.setAttribute('aria-hidden', 'true'); }
    });

    // Disable annotation editor mode via API
    try {
      this.app.pdfViewer.annotationEditorMode = { mode: -1 };
    } catch (_) {}
  }

  // ── Floating Toolbar ────────────────────────────────────────

  _buildFloatingToolbar() {
    this.floatingTb = T.floatingToolbar(this.toolMgr.allTools);
    document.body.appendChild(this.floatingTb);

    this.floatingTb.addEventListener('click', (e) => {
      const btn = e.target.closest('.ann-ftool-btn');
      if (!btn) return;
      this.setActiveTool(btn.dataset.tool);
    });
  }

  setActiveTool(id) {
    this._cancelForm(); // switching tools always cancels pending work
    this.toolMgr.setActive(id);
    this._syncToolbarUI();
    this._syncOverlayInteractivity();
  }

  _syncToolbarUI() {
    const activeId = this.toolMgr.activeTool?.id;
    this.floatingTb.querySelectorAll('.ann-ftool-btn').forEach(btn => {
      btn.classList.toggle('active', btn.dataset.tool === activeId);
    });
  }

  _syncOverlayInteractivity() {
    const active = this.toolMgr.isAnnotating;
    const overlayClass = this.toolMgr.overlayClass;
    this.overlays.forEach(ov => {
      ov.classList.toggle('active', active);
      // Remove all mode-* classes, then add the current one
      ov.className = ov.className.replace(/\bmode-\S+/g, '').trim();
      if (active && overlayClass) {
        ov.classList.add(overlayClass);
      }
    });
  }

  // ── Keyboard shortcuts ──────────────────────────────────────

  _bindKeyboard() {
    document.addEventListener('keydown', (e) => {
      if (e.target.tagName === 'TEXTAREA' || e.target.tagName === 'INPUT' || e.target.tagName === 'SELECT') return;

      for (const tool of this.toolMgr.tools) {
        if (!tool.shortcut) continue;
        if (tool.shortcut === 'escape' && e.key === 'Escape') {
          this.setActiveTool(tool.id);
          return;
        }
        if (e.key.toLowerCase() === tool.shortcut) {
          this.setActiveTool(tool.id);
          return;
        }
      }
    });
  }

  // ── Sidebar Pane (plugs into native sidebar) ────────────────
  //
  // pdf.js 4.1.392 sidebar structure:
  //   #sidebarContainer
  //     #toolbarSidebar          (sidebar header with view radio buttons)
  //       #sidebarViewButtons    (viewThumbnail, viewOutline, viewAttachments, viewLayers)
  //     #sidebarContent          (contains thumbnailView, outlineView, attachmentsView, layersView)
  //     #sidebarResizer
  //
  // The outer container gets class "sidebarOpen" when the sidebar is visible.
  // The native toggle button is #sidebarToggle.
  //

  _buildSidebarPane() {
    const container = document.getElementById('sidebarContent');
    if (!container) { console.warn('[AnnotationLayer] No sidebar container found'); return; }

    // Create our pane — starts hidden, matching the pattern of native views
    const pane = document.createElement('div');
    pane.id = 'ann-pane';
    pane.className = 'hidden';
    pane.innerHTML = T.paneHtml();
    container.appendChild(pane);

    this.paneEl = pane;
    this.listEl = pane.querySelector('.ann-pane-list');

    // Wire search and filter
    pane.querySelector('.ann-search-input').addEventListener('input', (e) => {
      this.textFilter = e.target.value.toLowerCase().trim();
      this._rebuildList();
    });
    pane.querySelector('.ann-filter-select').addEventListener('change', (e) => {
      this.categoryFilter = e.target.value;
      this._rebuildList();
    });

    // Add a "Comments" button to the sidebar view buttons radio group.
    // In 4.1.392 the view buttons live in #sidebarViewButtons.
    const viewButtons = document.getElementById('sidebarViewButtons');
    if (viewButtons) {
      const commentsBtn = document.createElement('button');
      commentsBtn.id = 'viewComments';
      commentsBtn.className = 'toolbarButton';
      commentsBtn.title = 'Show Comments';
      commentsBtn.tabIndex = 6;
      commentsBtn.setAttribute('role', 'radio');
      commentsBtn.setAttribute('aria-checked', 'false');
      commentsBtn.setAttribute('aria-controls', 'ann-pane');
      commentsBtn.innerHTML = `Comments`;
      commentsBtn.addEventListener('click', () => this._activateCommentsView());
      viewButtons.appendChild(commentsBtn);
      this._sidebarViewBtn = commentsBtn;
    }

    // Also add a dedicated toggle button in the main toolbar for quick access
    const btn = document.createElement('button');
    btn.id = 'ann-sidebar-btn';
    btn.innerHTML = T.BUBBLE_SVG;
    btn.title = 'Comments';
    btn.type = 'button';
    btn.addEventListener('click', () => this.togglePane());

    const toolbarLeft = document.getElementById('toolbarViewerLeft');
    const nativeToggle = document.getElementById('sidebarToggle');

    if (toolbarLeft && nativeToggle && nativeToggle.parentNode === toolbarLeft) {
      // Insert after sidebarToggle + spacer
      const spacer = nativeToggle.nextElementSibling;
      const ref = spacer && spacer.classList.contains('toolbarButtonSpacer')
        ? spacer.nextSibling
        : nativeToggle.nextSibling;
      toolbarLeft.insertBefore(btn, ref);
    } else if (toolbarLeft) {
      toolbarLeft.appendChild(btn);
    }
    this._sbBtn = btn;
  }

  /**
   * Activate the comments view within the native sidebar system.
   * Hides all native view divs, shows ours, updates button states,
   * and opens the sidebar if it's closed.
   */
  _activateCommentsView() {
    this.paneOpen = true;
    this._sbBtn?.classList.add('active');

    this._applyCommentsViewState();

    // If sidebar isn't open, open it via the native toggle.
    // The sidebarToggle click fires sidebarviewchanged synchronously
    // (switching to THUMBS). We suppress that with the _openingSidebar flag.
    const outer = document.getElementById('outerContainer');
    if (outer && !outer.classList.contains('sidebarOpen')) {
      const tog = document.getElementById('sidebarToggle');
      if (tog) {
        this._openingSidebar = true;
        tog.click();
        this._openingSidebar = false;
        // Re-apply after pdfjs switchView has run
        requestAnimationFrame(() => this._applyCommentsViewState());
      }
    }

    this._rebuildList();
  }

  /**
   * Set the sidebar content to show comments and nothing else.
   * Safe to call multiple times.
   */
  _applyCommentsViewState() {
    // Hide all native view content divs (4.1.392 ids)
    ['thumbnailView', 'outlineView', 'attachmentsView', 'layersView'].forEach(id => {
      const v = document.getElementById(id);
      if (v) v.classList.add('hidden');
    });

    // Deselect all native sidebar view buttons
    ['viewThumbnail', 'viewOutline', 'viewAttachments', 'viewLayers'].forEach(id => {
      const btn = document.getElementById(id);
      if (btn) {
        btn.classList.remove('toggled');
        btn.setAttribute('aria-checked', 'false');
      }
    });

    // Show our pane and select our button
    this.paneEl.classList.remove('hidden');
    if (this._sidebarViewBtn) {
      this._sidebarViewBtn.classList.add('toggled');
      this._sidebarViewBtn.setAttribute('aria-checked', 'true');
    }

    // Hide the outline options container (only relevant for outline view)
    const outlineOpts = document.getElementById('outlineOptionsContainer');
    if (outlineOpts) outlineOpts.style.display = 'none';
  }

  /**
   * Called when pdfjs closes the sidebar or switches to a native view.
   * We just mark ourselves as hidden — we never touch the sidebar toggle.
   */
  _onPaneClosed() {
    if (!this.paneOpen) return;
    this.paneOpen = false;
    this._cancelForm();
    this.paneEl.classList.add('hidden');
    this._sbBtn?.classList.remove('active');
    if (this._sidebarViewBtn) {
      this._sidebarViewBtn.classList.remove('toggled');
      this._sidebarViewBtn.setAttribute('aria-checked', 'false');
    }
    // Restore outline options visibility (it manages itself, but reset just in case)
    const outlineOpts = document.getElementById('outlineOptionsContainer');
    if (outlineOpts) outlineOpts.style.display = '';
  }

  /**
   * Open the comments pane (idempotent — does nothing if already open).
   */
  openPane() {
    if (this.paneOpen) return;
    this._activateCommentsView();
  }

  /**
   * Close the comments pane. Closes the entire sidebar since we
   * don't know which native view the user was looking at before.
   */
  closePane() {
    if (!this.paneOpen) return;
    this._onPaneClosed();
    // Close the sidebar entirely
    const outer = document.getElementById('outerContainer');
    if (outer && outer.classList.contains('sidebarOpen')) {
      const tog = document.getElementById('sidebarToggle');
      if (tog) {
        this._openingSidebar = true;
        tog.click();
        this._openingSidebar = false;
      }
    }
  }

  togglePane() {
    if (this.paneOpen) this.closePane(); else this.openPane();
  }

  // ── New-comment form (in-flow in the pane list) ─────────────

  showNewForm(opts) {
    if (this.pendingForm) {
      const ta = this.pendingForm.el.querySelector('textarea');
      if (ta && ta.value.trim()) {
        this.pendingForm.el.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        return;
      }
      this._cancelForm();
    }

    this._rebuildList();

    const formEl = T.newFormElement(opts);
    this._insertFormInFlow(formEl, opts.page, opts.y || 0);
    this.pendingForm = { el: formEl, opts };

    const textarea = formEl.querySelector('textarea');
    const catSelect = formEl.querySelector('.ann-new-form-cat');

    setTimeout(() => textarea.focus(), 60);

    formEl.querySelector('.cancel').addEventListener('click', () => this._cancelForm());
    formEl.querySelector('.save').addEventListener('click', () => {
      const cat = catSelect.value;
      if (!cat) { catSelect.classList.add('error'); catSelect.focus(); return; }
      const text = textarea.value.trim();
      if (!text) { textarea.focus(); return; }
      this._addAnnotation({
        type: opts.type, page: opts.page,
        x: opts.x, y: opts.y,
        width: opts.width || 0, height: opts.height || 0,
        text, category: cat,
      });
      this.pendingForm = null;
      formEl.remove();
    });

    textarea.addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); formEl.querySelector('.save').click(); }
      if (e.key === 'Escape') this._cancelForm();
    });
    catSelect.addEventListener('change', () => catSelect.classList.remove('error'));

    formEl.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  }

  _cancelForm() {
    if (this.pendingForm) {
      this.pendingForm.el.remove();
      this.pendingForm = null;
    }
  }

  _insertFormInFlow(formEl, page, y) {
    const list = this.listEl;
    const children = list.children;
    let curPage = -1;

    for (let i = 0; i < children.length; i++) {
      const ch = children[i];
      if (ch.classList.contains('ann-page-sep')) {
        const sepPage = parseInt(ch.textContent.replace(/\D/g, ''), 10);
        if (sepPage > page) {
          list.insertBefore(formEl, ch);
          return;
        }
        curPage = sepPage;
      } else if (ch.classList.contains('ann-card') && curPage === page) {
        const cardY = parseFloat(ch.dataset.annY || '0');
        if (cardY > y) {
          list.insertBefore(formEl, ch);
          return;
        }
      }
    }

    if (curPage !== page) {
      list.appendChild(T.pageSeparator(page));
    }
    list.appendChild(formEl);
  }

  // ── Pane list rendering (with page separators) ──────────────

  _getReplies(parentId) {
    const out = [];
    this.annotations.forEach(a => { if (a.parentId === parentId) out.push(a); });
    return out.sort((a, b) => (a.createdAt || '').localeCompare(b.createdAt || ''));
  }

  _rebuildList() {
    const list = this.listEl;
    list.innerHTML = '';
    this.cardElements.clear();

    const topLevel = [];
    this.annotations.forEach(a => { if (!a.parentId) topLevel.push(a); });
    topLevel.sort((a, b) => a.page !== b.page ? a.page - b.page : a.y - b.y);

    const filtered = topLevel.filter(a => {
      if (this.categoryFilter && a.category !== this.categoryFilter) return false;
      if (!this.textFilter) return true;
      const haystack = [
        a.text, a.author, a.category, 'page ' + a.page, a.type,
        ...this._getReplies(a.id).map(r => r.text + ' ' + r.author),
      ].join(' ').toLowerCase();
      return haystack.includes(this.textFilter);
    });

    if (!filtered.length) {
      list.innerHTML = `
${
        this.textFilter || this.categoryFilter
          ? 'No comments match your filters.'
          : 'No comments yet.
Use the toolbar to add comments or highlights.'
      }
`;
      if (this.pendingForm) this._insertFormInFlow(this.pendingForm.el, this.pendingForm.opts.page, this.pendingForm.opts.y || 0);
      return;
    }

    let curPage = -1;
    filtered.forEach(ann => {
      if (ann.page !== curPage) {
        curPage = ann.page;
        list.appendChild(T.pageSeparator(curPage));
      }
      const card = this._buildCard(ann);
      list.appendChild(card);
      this.cardElements.set(ann.id, card);
    });

    if (this.pendingForm) {
      this._insertFormInFlow(this.pendingForm.el, this.pendingForm.opts.page, this.pendingForm.opts.y || 0);
    }
  }

  _buildCard(ann) {
    const card = T.cardElement(ann, this.config.currentUser);
    if (ann.id === this.selectedId) card.classList.add('sel');

    card.addEventListener('click', (e) => {
      const action = e.target.closest('.ann-card-action');
      if (action) {
        e.stopPropagation();
        const act = action.dataset.action;
        if (act === 'reply') this._showReplyInput(ann.id, card);
        else if (act === 'edit') this._editAnnotation(ann.id);
        else if (act === 'delete') this._deleteAnnotation(ann.id);
        return;
      }
      if (e.target.closest('.ann-reply-input') || e.target.closest('textarea')) return;
      this._scrollToAnnotation(ann.id);
      this._selectAnnotation(ann.id);
    });

    const replies = this._getReplies(ann.id);
    if (replies.length) {
      const repliesDiv = document.createElement('div');
      repliesDiv.className = 'ann-replies';
      replies.forEach(r => repliesDiv.appendChild(T.replyElement(r, this.config.currentUser)));
      card.appendChild(repliesDiv);
    }

    return card;
  }

  _showReplyInput(parentId, card) {
    if (card.querySelector('.ann-reply-input')) return;
    let repliesDiv = card.querySelector('.ann-replies');
    if (!repliesDiv) {
      repliesDiv = document.createElement('div');
      repliesDiv.className = 'ann-replies';
      card.appendChild(repliesDiv);
    }
    const wrap = document.createElement('div');
    wrap.className = 'ann-reply-input';
    wrap.innerHTML = T.replyInputHtml();
    repliesDiv.appendChild(wrap);

    const ta = wrap.querySelector('.ann-reply-textarea');
    const sendBtn = wrap.querySelector('.ann-reply-send');
    setTimeout(() => ta.focus(), 30);

    const send = async () => {
      const text = ta.value.trim();
      if (!text) { ta.focus(); return; }
      const parent = this.annotations.get(parentId);
      await this._addAnnotation({
        type: 'comment', parentId,
        page: parent ? parent.page : 1,
        x: 0, y: 0, text, category: null,
      });
      wrap.remove();
    };

    sendBtn.addEventListener('click', send);
    ta.addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); send(); }
      if (e.key === 'Escape') wrap.remove();
    });
  }

  // ── Page overlays ───────────────────────────────────────────

  _ensureOverlays() {
    document.querySelectorAll('.page[data-page-number]').forEach(pageDiv => {
      const pageNum = parseInt(pageDiv.dataset.pageNumber, 10);

      if (this.overlays.has(pageNum)) {
        const existing = this.overlays.get(pageNum);
        if (!pageDiv.contains(existing)) {
          pageDiv.style.position = 'relative';
          pageDiv.appendChild(existing);
        }
        return;
      }

      const overlay = document.createElement('div');
      overlay.className = 'ann-overlay';
      overlay.dataset.annPage = pageNum;
      pageDiv.style.position = 'relative';
      pageDiv.appendChild(overlay);
      this.overlays.set(pageNum, overlay);

      this._bindOverlay(overlay, pageNum, pageDiv);
      this._renderAnnotationsForPage(pageNum);
      this._syncOverlayInteractivity();
    });
  }

  _bindOverlay(overlay, pageNum, pageDiv) {
    const makeCtx = (e) => ({
      pageNumber: pageNum,
      pageDiv,
      overlay,
      norm: pixelToNorm(pageDiv, e.clientX, e.clientY),
      layer: this,
    });

    overlay.addEventListener('click', (e) => {
      this.toolMgr.dispatch('onPageClick', makeCtx(e), e);
    });
    overlay.addEventListener('mousedown', (e) => {
      this.toolMgr.dispatch('onPageMouseDown', makeCtx(e), e);
    });
    overlay.addEventListener('mousemove', (e) => {
      this.toolMgr.dispatch('onPageMouseMove', makeCtx(e), e);
    });
    overlay.addEventListener('mouseup', (e) => {
      this.toolMgr.dispatch('onPageMouseUp', makeCtx(e), e);
    });
  }

  // ── Render annotations on pages ─────────────────────────────

  _renderAll() {
    this.annotations.forEach(a => { if (!a.parentId) this._renderAnnotation(a); });
  }

  _renderAnnotationsForPage(pageNum) {
    this.annotations.forEach(a => {
      if (a.page === pageNum && !a.parentId) this._renderAnnotation(a);
    });
  }

  _renderAnnotation(ann) {
    const existing = this.elements.get(ann.id);
    if (existing) existing.remove();

    const overlay = this.overlays.get(ann.page);
    if (!overlay) return;

    let el;
    if (ann.type === 'comment') {
      el = document.createElement('div');
      el.className = 'ann-pin';
      el.style.left = (ann.x * 100) + '%';
      el.style.top = (ann.y * 100) + '%';
      el.innerHTML = T.BUBBLE_SVG;
      el.title = ann.text || '';
    } else if (ann.type === 'highlight') {
      el = document.createElement('div');
      el.className = 'ann-hl';
      el.style.left = (ann.x * 100) + '%';
      el.style.top = (ann.y * 100) + '%';
      el.style.width = (ann.width * 100) + '%';
      el.style.height = (ann.height * 100) + '%';
      el.title = ann.text || '';
    }

    if (!el) return;

    el.dataset.annId = ann.id;
    if (ann.id === this.selectedId) el.classList.add('sel');

    el.addEventListener('click', (e) => {
      e.stopPropagation();
      this._selectAnnotation(ann.id);
      this.openPane();
      this._scrollToCard(ann.id);
    });

    overlay.appendChild(el);
    this.elements.set(ann.id, el);
  }

  // ── CRUD ────────────────────────────────────────────────────

  async _addAnnotation(data) {
    const user = this.config.currentUser;
    const ann = {
      id: genId(),
      type: data.type,
      parentId: data.parentId || null,
      page: data.page,
      x: data.x, y: data.y,
      width: data.width || 0, height: data.height || 0,
      text: data.text || '',
      category: data.category || null,
      author: user.name || 'Reviewer',
      authorInitials: user.initials || T.initials(user.name || 'Reviewer'),
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
      _persisted: false,
    };

    this.annotations.set(ann.id, ann);
    if (!ann.parentId) this._renderAnnotation(ann);
    this._rebuildList();

    try {
      const saved = await this.store.save(ann);
      if (saved?.id && saved.id !== ann.id) {
        this.annotations.delete(ann.id);
        ann.id = saved.id;
        this.annotations.set(ann.id, ann);
      }
      ann._persisted = true;
    } catch (e) {
      console.error('[AnnotationLayer] Save failed:', e);
    }

    if (!ann.parentId) {
      this._selectAnnotation(ann.id);
      setTimeout(() => this._scrollToCard(ann.id), 100);
    }
  }

  async _deleteAnnotation(id) {
    const ann = this.annotations.get(id);
    if (!ann) return;

    this._getReplies(id).forEach(r => {
      this.annotations.delete(r.id);
      this.store.remove(r.id).catch(() => {});
    });

    const el = this.elements.get(id);
    if (el) el.remove();
    this.elements.delete(id);
    this.annotations.delete(id);
    if (this.selectedId === id) this.selectedId = null;
    this._rebuildList();

    try { await this.store.remove(id); } catch (e) { console.error('[AnnotationLayer] Delete failed:', e); }
  }

  _editAnnotation(id) {
    const ann = this.annotations.get(id);
    if (!ann) return;
    const card = this.cardElements.get(id);
    if (!card) return;

    const textDiv = card.querySelector('.ann-card-text');
    textDiv.innerHTML = T.editFormHtml(ann.text || '');

    const ta = textDiv.querySelector('.ann-edit-textarea');
    ta.focus();
    ta.setSelectionRange(ta.value.length, ta.value.length);

    const finish = async (save) => {
      if (save) {
        ann.text = ta.value.trim();
        ann.updatedAt = new Date().toISOString();
        try { await this.store.save(ann); } catch (_) {}
      }
      this._rebuildList();
      if (!ann.parentId) this._renderAnnotation(ann);
    };

    textDiv.querySelector('.ann-edit-cancel').addEventListener('click', () => finish(false));
    textDiv.querySelector('.ann-edit-save').addEventListener('click', () => finish(true));
    ta.addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); finish(true); }
      if (e.key === 'Escape') finish(false);
    });
  }

  // ── Selection & navigation ──────────────────────────────────

  _selectAnnotation(id) {
    const prevId = this.selectedId;
    this.selectedId = id;

    [prevId, id].forEach(annId => {
      if (!annId) return;
      const el = this.elements.get(annId);
      if (el) el.classList.toggle('sel', annId === id);
      const card = this.cardElements.get(annId);
      if (card) card.classList.toggle('sel', annId === id);
    });
  }

  _scrollToAnnotation(id) {
    const ann = this.annotations.get(id);
    if (!ann) return;
    if (this.app?.pdfViewer) {
      this.app.pdfViewer.currentPageNumber = ann.page;
    }
    setTimeout(() => {
      const el = this.elements.get(id);
      if (el) el.scrollIntoView({ behavior: 'smooth', block: 'center' });
    }, 300);
  }

  _scrollToCard(id) {
    const card = this.cardElements.get(id);
    if (card) card.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  }

  // ── Viewer wait helpers ─────────────────────────────────────

  _waitForViewer() {
    return new Promise(resolve => {
      const check = () => {
        if (window.PDFViewerApplication?.pdfViewer) resolve(window.PDFViewerApplication);
        else requestAnimationFrame(check);
      };
      check();
    });
  }

  _waitForFirstPage() {
    return new Promise(resolve => {
      const check = () => {
        if (document.querySelector('.page[data-page-number]')) resolve();
        else requestAnimationFrame(check);
      };
      this.eventBus.on('pagesloaded', () => setTimeout(check, 50));
      check();
    });
  }

  // ── Internal cleanup ─────────────────────────────────────────

  _clearAll() {
    this._cancelForm();
    this.elements.forEach(el => el.remove());
    this.elements.clear();
    this.annotations.clear();
    this.cardElements.clear();
    this.selectedId = null;
    if (this.listEl) this.listEl.innerHTML = '';
  }

  // ── Public API ──────────────────────────────────────────────

  getAnnotations() {
    return [...this.annotations.values()].map(({ _persisted, ...rest }) => rest);
  }

  addAnnotation(data) {
    return this._addAnnotation(data);
  }

  destroy() {
    this.overlays.forEach(o => o.remove());
    this.overlays.clear();
    this.elements.clear();
    this.annotations.clear();
    document.getElementById('ann-layer-css')?.remove();
    this.paneEl?.remove();
    this.floatingTb?.remove();
    this._sbBtn?.remove();
    if (this._sidebarViewBtn) this._sidebarViewBtn.remove();
  }
}

export function restStorage(apiBaseUrl, documentId) {
  const base = apiBaseUrl.replace(/\/$/, '');
  const docPath = `${base}/${encodeURIComponent(documentId)}`;

  return {
    async load() {
      const r = await fetch(docPath);
      if (!r.ok) throw new Error(`Load failed: ${r.status}`);
      return r.json();
    },
    async save(annotation) {
      const method = annotation._persisted ? 'PUT' : 'POST';
      const url = annotation._persisted
        ? `${docPath}/${encodeURIComponent(annotation.id)}`
        : docPath;
      const r = await fetch(url, {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(annotation),
      });
      if (!r.ok) throw new Error(`Save failed: ${r.status}`);
      return r.json();
    },
    async remove(annotationId) {
      const r = await fetch(`${docPath}/${encodeURIComponent(annotationId)}`, { method: 'DELETE' });
      if (!r.ok) throw new Error(`Delete failed: ${r.status}`);
    },
  };
}

export function memoryStorage() {
  return {
    load: () => Promise.resolve([]),
    save: (a) => Promise.resolve(a),
    remove: () => Promise.resolve(),
  };
}

export function buildStorage(config) {
  if (config.onSave && config.onLoad) {
    return {
      load: config.onLoad.bind(null, config.documentId),
      save: config.onSave,
      remove: config.onDelete || (() => Promise.resolve()),
    };
  }
  if (config.documentId && config.apiBaseUrl) {
    return restStorage(config.apiBaseUrl, config.documentId);
  }
  return memoryStorage();
}

export function esc(s) {
  const d = document.createElement('div');
  d.textContent = s;
  return d.innerHTML;
}

export function initials(name) {
  return (name || 'R').split(/\s+/).map(w => w[0]).join('').slice(0, 2).toUpperCase();
}

export function formatDate(iso) {
  if (!iso) return '';
  const d = new Date(iso);
  return d.toLocaleDateString(undefined, { month: 'short', day: 'numeric', year: 'numeric' })
    + ' at '
    + d.toLocaleTimeString(undefined, { hour: '2-digit', minute: '2-digit' });
}

function catBadgeHtml(category) {
  const c = CATEGORY_COLORS[category];
  if (!c) return '';
  return `${esc(category)}`;
}

let _instance = null;

export const PDFAnnotationLayer = {
  async init(config = {}) {
    if (_instance) return _instance;
    const layer = new AnnotationLayer(config);
    await layer.init();
    _instance = layer;
    return layer;
  },
  getInstance() { return _instance; },
  destroy() {
    if (_instance) { _instance.destroy(); _instance = null; }
  },
};
